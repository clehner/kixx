 65 /** @private */
 66 TREADMILL.appendOutput = (function createAppendOutput() {
 67   /** @private */
 68   var that = function appendOutput(level, html) {
 69     var div = document.createElement("div");
 70     if (level === "failed") {
 71       div.setAttribute("class", "block-failed");
 72     } else if(level === "passed") {
 73       div.setAttribute("class", "block-passed");
 74     }
 75     document.getElementById("treadmill").appendChild(div);
 76     div.innerHTML = html;
 77   };
 78 
 79   /** @private */
 80   that.passed = function failed(html) {
 81     that("passed", html);
 82   };
 83 
 84   /** @private */
 85   that.failed = function passed(html) {
 86     that("failed", html);
 87   };
 88 
 89   return that;
 90 }());

 92 /**
 93  * @field
 94  * @description The standard Treadmill output formatter.
 95  * Outputs Simpletest results to the test document.
 96  * @example
 97  * var TestSuite = require("simpletest/testrunner_1").TestSuite;
 98  * var suite = new TestSuite("My Tests", TREADMILL.testOutputFormatter);
 99  */
100 TREADMILL.testOutputFormatter =
101 function createTestOutputFormatter(stdout) {
102   // public methods
103   return {
104     startSuite: function startSuite(name) {
105     },
106     endSuite: function endSuite(results) {
107       var passed = true, report;
108 
109       function arrayMap(f, l) {
110         var i, str = "";
111         for (i = 0; i < l.length; i += 1) {
112           str += f(l[i]);
113         }
114         return str;
115       }
116 
117       function dictMap(f, d) {
118         var p, str = "";
119         for (p in d) {
120           if (d.hasOwnProperty(p)) {
121             str += f(p, d[p]);
122           }
123         }
124         return str;
125       }
126 
127       function points(point) {
128         if (point.result !== "ok" || point.directive) {
129           return '<li>'+
130             point.num +' '+
131             point.result +': '+
132             point.description +' '+
133             point.directive +' '+ point.reason +
134             (point.result === "ok" ? '' : '<br /># '+ point.diagnostic) +
135             '</li>';
136         }
137         return '';
138       }
139 
140       function tests(test) {
141         var str = '<h5>'+ test.description +'</h5>',
142             testpoints;
143 
144         if (test.result === "not_ok") { // something in the test failed
145           passed = false;
146           return (str += 'failed test points:<ul>'+ arrayMap(points, test.points) +'</ul>');
147         }
148         if (test.result === "timed_out") {
149           passed = false;
150           return (str += 'timed out');
151         }
152         if (test.result === "error") {
153           passed = false;
154           return (str += '<pre>name: '+ test.error.name +
155               "\nmessage: "+ test.error.message +
156               "\nfile: "+ test.error.fileName +
157               "\nline: "+ test.error.lineNumber +'</pre>');
158         }
159         testpoints = arrayMap(points, test.points);
160         if (testpoints) {
161           return (str += 'directed test points:<ul>'+ testpoints +'</ul>');
162         }
163         return (str += 'passed');
164       }
165 
166       function cases(casename, cas) {
167         return '<h4>Testcase: '+ casename +'</h4>'+ arrayMap(tests, cas.tests);
168       }
169 
170       report = dictMap(cases, results.cases);
171       if (passed) {
172         stdout.passed(report);
173       } else {
174         stdout.failed(report);
175       }
176     },
177     startCase: function startCase(name) {
178     },
179     endCase: function endCase(result) {
180     },
181     startTest: function startTest(desc) {
182     },
183     endTest: function endTest(result) {
184     },
185     testpoint: function testpoint(result) {
186     }
187   };
188 };

190 /**
191  * Scan a JavaScript file with JSLint.
192  * <a href="http://jslint.com/" target="_blank">(JSLint.com)</a>
193  * Will output the results as an html snippet onto your testrunner page.
194  * Important!: To use this feature, you need to include
195  * <code>
196  * <script src="chrome://kixx/content/packs/jslint/fulljslint.js"
197  * ></script>
198  * </code>
199  *
200  * @param {string} aFile The location of the file using the same annotation
201  * as <code>require()</code>.
202  */
203 TREADMILL.jslint = function jslint(aFile) {
204   var url, result;
205 
206   if (typeof JSLINT !== "function") {
207     throw new Error("chrome://kixx/content/packs/jslint/fulljslint.js "+
208         "must be included with a script tag for JSLint to work. ");
209   }
210 
211   url = require.loader.normalize(require.loader.resolve(aFile.slice(0, -3)));
212   result = JSLINT(require.loader.fetch(url));
213   TREADMILL.appendOutput(
214       (result ? "passed" : "failed"), JSLINT.report(false));
215 };

217 /**
218  * Create a Simpletest test suite and run it. The output from the tests will be
219  * written to the container element with id "treadmill" when the suite is done
220  * running. This is just a convenience wrapper for the
221  * <code>require("simpletest/testrunner_1").TestSuite</code>
222  * constructor.
223  *
224  * @param {string} aName The name of the test group 
225  * @param {object[]} aTests A list of objects with the following members:
226  *  <br />{function} test The test function (will be passed the test
227  *  object)
228  *  <br />{integer} time The time limit for the test before it times out
229  *  in milliseconds
230  * @param {function} aCallback Will be called when the test suite is done running.
231  */
232 TREADMILL.createAndRunTestSuite = function createTestSuite(aName, aTests, aCallback) {
233   var simpletest = require("simpletest/testrunner_1"),
234       suite, i;
235   suite = new simpletest.TestSuite(
236       "Simpletest", TREADMILL.testOutputFormatter(TREADMILL.appendOutput));
237   for(i = 0; i < aTests.length; i += 1) {
238     aTests[i].time = aTests[i].time || 500;
239     suite.addTest(aName, aTests[i].test, aTests[i].test.name, null, aTests[i].time);
240   }
241   suite.run(aCallback);
242 };
